jQuery.basicTicker()

Simple CallBack
            jQuery("#basicTicker1").basicTicker(["Some text 1", "Some text 2", "Some text 3"], 1500, function(ele) {
                jQuery(ele).html("<span class='done'>All done: basicTicker1</span>").show(0);
            });
Event Binding
            jQuery("#basicTicker2").basicTicker(["Some text 1", "Some text 2", "Some text 3"], 2000);
            jQuery("#basicTicker2").bind("ticker:done", function(e) {
                jQuery(e.target).html("<span class='done'>All done: basicTicker2</span>").show(0);
            });
